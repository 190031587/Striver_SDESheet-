//set Matrix zeros --Arrays I

#include <bits/stdc++.h> 
void setZeros(vector<vector<int>> &a)
{
	int n=a.size();
    int m=a[0].size();
    int first[n],second[m];
    for(int i=0;i<n;i++)first[i]=-1;
    for(int i=0;i<m;i++)second[i]=-1;
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<m;j++)
        {
            if(a[i][j]==0)
            {
                first[i]=0;
                second[j]=0;
            }
        }
    }
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<m;j++)
        {
            if(first[i]==0||second[j]==0)
                a[i][j]=0;
        }
    }
    
}
